## 1. 测试策略

### 1.1 测试目标

- ✅ 验证功能完整性：确保所有需求功能正确实现
- ✅ 验证功能正确性：确保所有功能在各种场景下正常工作
- ✅ 验证非功能需求：性能、可靠性、易用性、兼容性、安全性
- ✅ 发现和修复缺陷：在发布前发现并修复尽可能多的Bug
- ✅ 提升用户体验：确保界面友好、操作流畅、提示清晰

### 1.2 测试原则

1. **测试驱动开发**：核心业务逻辑模块先编写单元测试
2. **分层测试**：单元测试 → 集成测试 → 系统测试 → 验收测试
3. **自动化优先**：单元测试和集成测试必须自动化
4. **人工补充**：UI测试、兼容性测试以人工测试为主
5. **回归测试**：每次修复Bug后执行回归测试
6. **性能基准**：建立性能基准，持续监控性能指标

### 1.3 测试范围

| 测试范围 | 包含内容 | 排除内容 |
| ----- | ---- | ---- |
| **功能测试** | 所有需求文档中定义的功能 | 第三方库内部逻辑 |
| **性能测试** | 启动时间、内存占用、CPU占用、UI响应速度 | 第三方库性能 |
| **兼容性测试** | Windows 10/11、多显示器、高DPI | macOS、Linux |
| **安全性测试** | 密码加密存储、输入验证、异常处理 | 渗透测试、压力测试 |
| **易用性测试** | 界面友好性、操作流畅性、提示清晰度 | 无障碍测试 |

```csharp
// ConfigServiceTests.cs
public class ConfigServiceTests
{
    [Fact]
    public void LoadConfig_WithValidConfig_ReturnsConfig()
    {
        // Arrange
        var configPath = "testdata/valid_config.json";
        var service = new ConfigService(configPath);

        // Act
        var config = service.LoadConfig();

        // Assert
        config.Should().NotBeNull();
        config.RestIntervalMinutes.Should().Be(30);
        config.RestDurationSeconds.Should().Be(20);
    }

    [Fact]
    public void LoadConfig_WithCorruptedConfig_ReturnsDefaultConfig()
    {
        // Arrange
        var configPath = "testdata/corrupted_config.json";
        var service = new ConfigService(configPath);

        // Act
        var config = service.LoadConfig();

        // Assert
        config.Should().NotBeNull();
        config.RestIntervalMinutes.Should().Be(30); // 默认值
    }

    [Theory]
    [InlineData(0, false)]  // 小于最小值
    [InlineData(1, true)]   // 等于最小值
    [InlineData(120, true)] // 等于最大值
    [InlineData(121, false)] // 大于最大值
    public void ValidateConfig_WithRestInterval_ReturnsExpectedResult(int value, bool isValid)
    {
        // Arrange
        var service = new ConfigService();
        var config = new AppConfig { RestIntervalMinutes = value };

        // Act
        var result = service.ValidateConfig(config);

        // Assert
        result.Should().Be(isValid);
    }
}
```

#### 3.2.2 集成测试示例

```csharp
// TimerIntegrationTests.cs
public class TimerIntegrationTests : IAsyncLifetime
{
    private TimerService _timerService;
    private MainViewModel _mainViewModel;

    public async Task InitializeAsync()
    {
        _timerService = new TimerService();
        _mainViewModel = new MainViewModel(_timerService, null);
        await Task.CompletedTask;
    }

    public async Task DisposeAsync()
    {
        _timerService?.Cleanup();
        await Task.CompletedTask;
    }

    [Fact]
    public void MainCountdownElapsed_ShouldShowRestWindow()
    {
        // Arrange
        var restWindowShown = false;
        _mainViewModel.RestWindowShown += (s, e) => restWindowShown = true;

        // Act
        _timerService.StartMainCountdown(TimeSpan.FromSeconds(2));
        // 模拟等待2秒...
        Thread.Sleep(2100);

        // Assert
        restWindowShown.Should().BeTrue();
        _timerService.State.Should().Be(TimerState.Resting);
    }

    [Fact]
    public void TrayPauseResume_ShouldSyncMainWindowState()
    {
        // Arrange
        _timerService.StartMainCountdown(TimeSpan.FromMinutes(30));

        // Act - 从托盘暂停
        _mainViewModel.PauseResumeFromTrayCommand.Execute(null);

        // Assert
        _timerService.State.Should().Be(TimerState.Paused);
        _mainViewModel.PauseResumeButtonText.Should().Be("恢复提醒");
        _mainViewModel.StatusText.Should().Be("已暂停");

        // Act - 从托盘恢复
        _mainViewModel.PauseResumeFromTrayCommand.Execute(null);

        // Assert
        _timerService.State.Should().Be(TimerState.Running);
        _mainViewModel.PauseResumeButtonText.Should().Be("暂停提醒");
        _mainViewModel.StatusText.Should().Be("运行中");
    }
}
```

#### 3.4.2 性能测试示例

```csharp
// PerformanceTests.cs
[MemoryDiagnoser]
public class PerformanceTests
{
    [Benchmark]
    public void ApplicationStartup()
    {
        var app = new App();
        app.InitializeComponent();
        // 测量启动时间
    }

    [Fact]
    public void CountdownAccuracy_ShouldBeWithin100ms()
    {
        // Arrange
        var timerService = new TimerService();
        var stopwatch = Stopwatch.StartNew();
        var targetDuration = TimeSpan.FromSeconds(5);

        // Act
        timerService.StartMainCountdown(targetDuration);

        // 模拟等待...
        Thread.Sleep(5000);

        stopwatch.Stop();

        // Assert
        var actualElapsed = stopwatch.Elapsed;
        var deviation = Math.Abs((actualElapsed - targetDuration).TotalMilliseconds);

        deviation.Should().BeLessThan(100); // ±100ms
    }
}
```
